    # 提取各方判定
    precheck_score = 1.0 if precheck_result.get("info_sufficiency", {}).get("is_sufficient") else 0.0
    phase1_score = 1.0 if phase1_result.get("info_status") == "sufficient" else 0.0
    mode_score = 1.0 if mode_adjustment_result.get("adjusted_status") == "sufficient" else 0.0

    # 加权投票
    final_score = precheck_score * 0.4 + phase1_score * 0.4 + mode_score * 0.2
