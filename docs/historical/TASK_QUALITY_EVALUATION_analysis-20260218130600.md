2. **添加任务质量评分机制**
```python
def _evaluate_task_quality(task: Dict) -> float:
    """评估单个任务的质量分数(0-1)"""
    score = 1.0

    # 检查标题长度（理想15-30字）
    title_len = len(task['title'])
    if title_len > 35:
        score -= 0.2  # 可能过于复杂

    # 检查多主体问题
    subject_count = task['title'].count('与') + task['title'].count('及')
    if subject_count >= 2:
        score -= 0.3  # 主体过多

    # 检查宏大词汇
    broad_keywords = ["振兴", "战略", "智慧", "逻辑", "体系"]
    for kw in broad_keywords:
        if kw in task['title']:
            score -= 0.2

    return max(score, 0.0)
```

```python
class TaskHierarchy:
    """任务层级管理"""
    def __init__(self):
        self.root_tasks = []  # 一级任务
        self.sub_tasks = {}   # 二级任务

    def decompose_broad_task(self, task: Dict) -> List[Dict]:
        """将宏大任务分解为子任务"""
        if self._is_too_broad(task):
            return self._generate_subtasks(task)
        return [task]

    def _is_too_broad(self, task: Dict) -> bool:
        """判断任务是否过于宏大"""
        broad_indicators = [
            len(task['title']) > 30,

        ]
        return sum(broad_indicators) >= 2
```
